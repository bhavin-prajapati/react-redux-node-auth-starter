import * as page from '../actions/page';
 
export const login = () => {
  return (dispatch) => {
    dispatch(page.login());
 
    // Login call is made here and success after 2 seconds
    setTimeout(() => {
      dispatch(page.loginSuccess());
    }, 2000);
  };
};
